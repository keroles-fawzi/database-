🔧 Lab 3 — ALTER TABLE & Keys
تعديل هيكل الجداول وتعريف المفاتيح
➕
ADD COLUMN — إضافة عمود جديد
ALTER TABLE ... ADD
ALTER
▼
الصيغة العامة
ALTER TABLE table_name ADD column_name datatype;
مثال — إضافة عمود email
ALTER TABLE student ADD email VARCHAR2(100);
يضيف العمود للجدول دون حذف أي بيانات موجودة
القيمة الافتراضية للعمود الجديد ستكون NULL لجميع الصفوف الموجودة
✏️
MODIFY COLUMN — تعديل عمود موجود
ALTER TABLE ... MODIFY
ALTER
▼
تغيير حجم العمود
ALTER TABLE student MODIFY student_name VARCHAR2(100);
إضافة قيد NOT NULL
ALTER TABLE student MODIFY student_name VARCHAR2(50) NOT NULL;
تصغير الحجم قد يفشل إذا كانت البيانات الموجودة أكبر من الحجم الجديد
🗑️
DROP COLUMN — حذف عمود
ALTER TABLE ... DROP COLUMN
خطر
▼
ALTER TABLE student DROP COLUMN email;
العملية لا رجعة منها — البيانات تُحذف نهائياً مع العمود
لا يمكن حذف عمود إذا كان جزء من PRIMARY KEY أو FOREIGN KEY مباشرة
🔄
RENAME COLUMN — إعادة تسمية
ALTER TABLE ... RENAME COLUMN
ALTER
▼
ALTER TABLE student RENAME COLUMN student_name TO full_name;
يغير الاسم فقط دون المساس بالبيانات أو الـ datatype
🔑
PRIMARY KEY — المفتاح الأساسي
CONSTRAINT ... PRIMARY KEY
KEY
▼
تعريف PRIMARY KEY عند إنشاء الجدول
CREATE TABLE student ( student_id NUMBER, student_name VARCHAR2(50), CONSTRAINT student_pk PRIMARY KEY(student_id) );
PRIMARY KEY = NOT NULL + UNIQUE — لا يمكن أن يكون NULL أو مكرر
اسم الـ constraint (student_pk) اختياري لكن من الأفضل تسميته
إضافة PRIMARY KEY لجدول موجود
ALTER TABLE student ADD CONSTRAINT student_pk PRIMARY KEY(student_id);
🔗
FOREIGN KEY — المفتاح الأجنبي
CONSTRAINT ... FOREIGN KEY ... REFERENCES
KEY
▼
الخطوة 1 — إنشاء جدول الأب (department) أولاً
CREATE TABLE department ( dept_id NUMBER, dept_name VARCHAR2(30), CONSTRAINT dept_pk PRIMARY KEY(dept_id) );
الخطوة 2 — إنشاء جدول الابن (student) مع FOREIGN KEY
CREATE TABLE student ( student_id NUMBER, student_name VARCHAR2(50), dept_id NUMBER, CONSTRAINT student_pk PRIMARY KEY(student_id), CONSTRAINT dept_fk FOREIGN KEY(dept_id) REFERENCES department(dept_id) );
يجب إنشاء جدول department أولاً — وإلا سيفشل الأمر
لا يمكن إدخال dept_id في student إذا لم تكن قيمته موجودة في department
عمود الـ FK في الابن يجب أن يكون نفس الـ datatype لعمود PK في الأب
📊 Lab 4 — DML Statements
أوامر التعامل مع البيانات
➕
INSERT — إدخال بيانات
INSERT INTO ... VALUES
DML
▼
إدخال كامل (كل الأعمدة بالترتيب)
INSERT INTO department VALUES(1, 'IT'); INSERT INTO department VALUES(2, 'CS'); INSERT INTO student VALUES(101, 'Ahmad', 1); INSERT INTO student VALUES(102, 'Sara', 2);
إدخال أعمدة محددة فقط
INSERT INTO student(student_id, student_name) VALUES(103, 'Omar'); -- dept_id ستكون NULL
يجب إدخال بيانات جدول الأب (department) أولاً قبل الابن (student)
الترتيب في VALUES يجب أن يطابق ترتيب الأعمدة في الجدول أو في قائمة الأعمدة
🔍
SELECT — استعلام البيانات
SELECT ... FROM ... WHERE
DML
▼
عرض كل البيانات
SELECT * FROM student;
عرض أعمدة محددة
SELECT student_name, dept_id FROM student;
عرض مع شرط WHERE
SELECT * FROM student WHERE student_id = 101;
JOIN بين جدولين — المثال الكامل
SELECT student_name, dept_name FROM student JOIN department ON student.dept_id = department.dept_id;
الـ JOIN يربط الجدولين بناءً على العمود المشترك (dept_id)
ORDER BY ترتيب النتائج
SELECT * FROM student ORDER BY student_name ASC; -- أو DESC
✏️
UPDATE — تعديل بيانات موجودة
UPDATE ... SET ... WHERE
DML
▼
تعديل صف واحد
UPDATE student SET student_name = 'Mohammad' WHERE student_id = 101;
تعديل أكثر من عمود
UPDATE student SET student_name = 'Mohammad', dept_id = 2 WHERE student_id = 101;
⚠️ بدون WHERE سيتم تعديل جميع الصفوف في الجدول!
-- هذا يعدل كل الطلاب: UPDATE student SET dept_id = 1; -- خطر!
🗑️
DELETE — حذف بيانات
DELETE FROM ... WHERE
خطر
▼
حذف صف معين
DELETE FROM student WHERE student_id = 102;
بدون WHERE يحذف كل البيانات من الجدول!
مقارنة DELETE vs TRUNCATE
-- DELETE: مع WHERE، يمكن Rollback DELETE FROM student WHERE student_id = 102; -- TRUNCATE: يحذف الكل، لا يمكن Rollback، أسرع TRUNCATE TABLE student;
الخاصيةDELETETRUNCATE
WHERE✓ يدعم✗ لا يدعم
Rollback✓ ممكن✗ مستحيل
السرعةأبطأأسرع
الـ Triggers✓ يفعّلها✗ لا يفعّلها
🔒 Constraints — قيود البيانات
ضمان صحة وسلامة البيانات
📋
مقارنة جميع أنواع القيود
▼
القيديمنعملاحظة مهمة
NOT NULLقيمة فارغة (NULL)الحقل إلزامي
UNIQUEتكرار القيمةيسمح بـ NULL واحد فقط
CHECKقيمة لا تحقق الشرطمثل salary > 0
PRIMARY KEYNULL والتكرار معاً= NOT NULL + UNIQUE
FOREIGN KEYقيمة غير موجودة في الأبيربط جدولين
🔵
NOT NULL
منع القيم الفارغة
▼
CREATE TABLE student ( student_id NUMBER, student_name VARCHAR2(50) NOT NULL, -- إلزامي email VARCHAR2(100) -- اختياري (NULL مسموح) );
يمنع إدخال صف بدون قيمة في هذا العمود
💚
UNIQUE
منع التكرار
▼
email VARCHAR2(50) UNIQUE -- أو بشكل constraint مسمى: CONSTRAINT email_uq UNIQUE(email)
يسمح بـ NULL ولكن مرة واحدة فقط — لأن NULL != NULL في SQL
✅
CHECK
شرط مخصص على القيم
▼
-- راتب موجب salary NUMBER CHECK (salary > 0) -- عمر بين 18 و 65 age NUMBER CHECK (age >= 18 AND age <= 65) -- constraint مسمى CONSTRAINT sal_check CHECK(salary > 0)
يمكن استخدام AND و OR و IN في شرط الـ CHECK
🏆
جدول كامل بجميع القيود
مثال شامل
▼
CREATE TABLE employee ( emp_id NUMBER, emp_name VARCHAR2(50) NOT NULL, email VARCHAR2(100) UNIQUE, salary NUMBER CHECK(salary > 0), dept_id NUMBER, CONSTRAINT emp_pk PRIMARY KEY(emp_id), CONSTRAINT emp_dept_fk FOREIGN KEY(dept_id) REFERENCES department(dept_id) );

📋 Cheatsheet سريع
مرجع سريع لجميع الأوامر والمفاهيم
🔧 ALTER TABLE
ALTER TABLE t ADD col type; ALTER TABLE t MODIFY col type; ALTER TABLE t DROP COLUMN col; ALTER TABLE t RENAME COLUMN old TO new;
🔑 Keys
-- PRIMARY KEY CONSTRAINT pk PRIMARY KEY(id) -- FOREIGN KEY CONSTRAINT fk FOREIGN KEY(col) REFERENCES parent(col)
📊 DML
INSERT INTO t VALUES(...); SELECT * FROM t; UPDATE t SET col=val WHERE ...; DELETE FROM t WHERE ...;
🔒 Constraints
col type NOT NULL col type UNIQUE col type CHECK(col > 0) PRIMARY KEY(col) FOREIGN KEY(col) REFERENCES t2(col)
📐 Relational Algebra
σ condition (R) = WHERE π cols (R) = SELECT cols R ∪ S = UNION R − S = MINUS R × S = CROSS JOIN R ⋈ S = JOIN
⚡ قواعد ذهبية
• PK = NOT NULL + UNIQUE
• أنشئ الأب قبل الابن
• UPDATE/DELETE بدون WHERE = خطر
• TRUNCATE لا رجعة منه
• σ = صفوف | π = أعمدة
🔍 مثال متكامل — من البداية للنهاية
-- 1. إنشاء جدول الأب CREATE TABLE department ( dept_id NUMBER, dept_name VARCHAR2(30) NOT NULL, CONSTRAINT dept_pk PRIMARY KEY(dept_id) ); -- 2. إنشاء جدول الابن CREATE TABLE student ( student_id NUMBER, student_name VARCHAR2(50) NOT NULL, email VARCHAR2(100) UNIQUE, dept_id NUMBER, CONSTRAINT student_pk PRIMARY KEY(student_id), CONSTRAINT dept_fk FOREIGN KEY(dept_id) REFERENCES department(dept_id) ); -- 3. إدخال بيانات INSERT INTO department VALUES(1, 'IT'); INSERT INTO student VALUES(101, 'Ahmad', 'a@uni.edu', 1); -- 4. استعلام مع JOIN SELECT student_name, dept_name FROM student JOIN department ON student.dept_id = department.dept_id; -- 5. تعديل UPDATE student SET student_name = 'Ahmad Ali' WHERE student_id = 101; -- 6. حذف DELETE FROM student WHERE student_id = 101;
